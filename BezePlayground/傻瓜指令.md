C:\CodexLab\codex\codex-rs\analytics\src\client.rs

```
fn try_send(&self, _input: AnalyticsFact) {    // Analytics disabled in secure build.}
```

```
fn try_send(&self, input: AnalyticsFact) {    #[cfg(test)]    {        let _ = self.sender.try_send(input);    }    #[cfg(not(test))]    {        let _ = input;        // Analytics disabled in secure build.    }}
```
